# Cheatsheet 

# List partitions inside a disk image
fdisk -l usb.image

# Attach disk image to a loop device (with partitions)
sudo losetup -fP usb.image
lsblk | grep loop

# Mount loop partition read-only
sudo mkdir -p /mnt/usb_p1
sudo mount -o ro /dev/loop0p1 /mnt/usb_p1

# If no partitions: mount whole image (FAT / simple FS)
sudo mount -o ro,loop usb.image /mnt/usb_p1

# If partition starts at offset:
sudo mount -o ro,loop,offset=$((SECTOR*512)) usb.image /mnt/usb_p1

# Unmount + detach loop
sudo umount /mnt/usb_p1
sudo losetup -d /dev/loop0d

# Sleuth Kit (TSK) 

# List files (recursive)
fls -r usb.image

# Extract file by inode
icat usb.image <inode> > output.bin

# Inspect inode metadata
istat usb.image <inode>

# File & Image Analysis 

# Inspect metadata
exiftool file.png

# Extract printable strings
strings file.png

# KeePass (.kdbx) Analysis 

# Show database header info
keepassxc-cli info Database.kdbx

# Convert header to readable fields (no cracking)
keepass2john Database.kdbx
